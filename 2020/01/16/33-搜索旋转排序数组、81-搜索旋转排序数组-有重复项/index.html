区别

两道题的区别是有无重复项，第一道中的if(nums[start] <= nums[mid])被拆分成if(nums[start] == nums[mid]){//相当于去掉一个重复的干扰项 start++; continue; }和if(nums[start] < nums[mid])

33. 搜索旋转排序数组

class Solution {
    public int search(int[] nums, int target) {
        if(nums == null || nums.length == 0)
            return -1;
        int start = 0;
        int end = nums.length - 1;
        int mid;
        while(start <= end){    
            mid = start + (end-start)/2;    
            if(nums[mid] == target)
                return mid;
            if(nums[start] <= nums[mid]){//等于是为了防止出现最后两项与逻辑不符的情况
                if(nums[start] <= target && nums[mid] > target){
                    end = mid - 1;
                }
                else{
                    start = mid + 1;
                }        
            }
            else{
                if(nums[mid] < target && nums[end] >= target){
                    start = mid + 1;
                }
                else{
                    end = mid - 1;
                }
            }
        }
        return -1;
    }
}

81. 搜索旋转排序数组 II（重复元素）

class Solution {
    public boolean search(int[] nums, int target) {
        if(nums == null || nums.length == 0)
            return false;
        int start = 0;
        int end = nums.length - 1;
        int mid;
        while(start <= end){    
            mid = start + (end-start)/2;    
            if(nums[mid] == target)
                return true;
            if(nums[start] == nums[mid]){//相当于去掉一个重复的干扰项。
                start++;
                continue;
            }
            if(nums[start] < nums[mid]){//这里不用=，因为前面已经判断过了
                if(nums[start] <= target && nums[mid] > target){
                    end = mid - 1;
                }
                else{
                    start = mid + 1;
                }        
            }
            else{
                if(nums[mid] < target && nums[end] >= target){
                    start = mid + 1;
                }
                else{
                    end = mid - 1;
                }
            }
        }
        return false;
    }
}
